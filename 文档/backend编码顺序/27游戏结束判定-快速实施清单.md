# 游戏结束判定 - 快速实施检查清单

> 基于《27游戏结束判定.md》的实施清单，包含所有代码改动要点

## 快速诊断

**当前问题症状:**
```
用户输入: "我猜是艺术史"
预期: {status: "finished", result: "user_win"}
实际: {status: "active", phase: "awaiting_answer", message: "请回答..."}
```

**根本原因:**
1. ❌ Intent 识别失败或识别为 "answer" 而非 "guess"
2. ❌ 即使识别为 "guess"，某些阶段拒绝了猜测
3. ❌ `_handle_user_guess()` 没有完整实现游戏结束逻辑

@classmethod
def _extract_guess_word(cls, normalized_text: str, markers: tuple) -> str:
    """从用户输入中提取猜测词汇"""
    result = normalized_text
    for marker in markers:
        if marker in result:
            result = result.replace(marker, "", 1).strip()
            break
    # 移除尾部标点
    result = re.sub(r"[。！？!?~\s]+$", "", result)
    return result

# ========== 修改 _infer_local_intent ==========

# 新代码:
@staticmethod
def _is_turn_allowed(phase: str, intent: str) -> bool:
    """任何阶段都允许用户猜测（除非游戏已结束）"""
    # ✅ 优先：任何非终止阶段都允许猜测
    if intent == "guess":
        return phase != "finished"
    
    # 其他意图的阶段限制
    if phase == "awaiting_answer":
        return intent in {"answer", "guess"}
    
    if phase == "awaiting_judgement":
        return intent in {"judge", "answer", "guess"}
    
    if phase == "user_turn":
        return intent in {"question", "guess"}
    
    return False

# 新代码结构:
async def submit_turn(self, game_id: str, question: str, mode: str = "agent", turn_type: str = "input"):
    # 1️⃣ 基本检查
    game = await self.repository.get_game(game_id)
    if not game:
        raise ValueError(...)
    if game.get("status") == "finished":
        raise ValueError(...)
    
    # 2️⃣ 检查轮数上限 ← 新增
    max_rounds = 20
    if game.get("round_count", 0) >= max_rounds:
        now = datetime.now()
        await self.repository.finish_game(game_id, "max_rounds_reached", now)
        return {
            "game_id": game_id,
            "status": "finished",
            "result": "draw",
            "message": f"已达最大轮数({max_rounds})，游戏结束",
        }
    
    # 3️⃣ 解析意图
    metadata = game.get("metadata", {}) or {}
    phase = metadata.get("phase", "user_turn")
    history = game.get("history", [])
    
    intent_result = await self.agent.parse_user_intent(question, history)
    intent = intent_result.get("intent", "invalid")
    
    # 4️⃣ 轮次锁校验（改进版）
    if not self._is_turn_allowed(phase, intent):
        return self._reject_turn(game_id, phase, 
                                f"当前阶段({phase})不允许此操作({intent})", history)
    
    # 5️⃣ 优先处理猜测 ← 关键改动：所有阶段都优先处理 guess
    if intent == "guess":
        return await self._handle_user_guess(game, intent_result)
    
    # 6️⃣ 处理其他意图（根据阶段）
    if phase == "awaiting_judgement":
        return await self._handle_agent_guess_judgement(game, question)
    
    if phase == "awaiting_answer":
        return await self._handle_user_answer(game, question, history, intent_result)
    
    if phase == "user_turn":
        if intent == "invalid":
            return self._reject_turn(game_id, phase, "请输入有效问题或猜测", history)
        if intent == "question":
            return await self._handle_user_question(game, question, history, mode)
        if intent == "answer":
            return self._reject_turn(game_id, phase, "现在轮到你提问或猜词", history)
    
    return self._reject_turn(game_id, phase, "游戏状态异常", history)
```

**关键改动:**
- ✅ 新增轮数上限检查（20 轮）
- ✅ 重构条件顺序：优先处理 "guess" intent
- ✅ 改进 `_is_turn_allowed()` 调用结果处理
- ✅ 简化阶段特定的处理逻辑

async def _handle_user_guess(self, game: dict, intent_result: dict):
    """处理用户猜测 - 完整版"""
    game_id = game["game_id"]
    user_guess = intent_result.get("guess_word", "").strip()
    current_phase = game.get("metadata", {}).get("phase", "user_turn")
    
    # 1️⃣ 验证猜测词不为空
    if not user_guess:
        return self._reject_turn(game_id, current_phase,
                                "猜测词不能为空，请输入有效猜测", 
                                game.get("history", []))
    
    # 2️⃣ 解密系统词
    try:
        target_word = self.reveal_target_word(game, self.settings)
    except Exception as e:
        logger.error("解密系统词失败 [game_id=%s]: %s", game_id, e)
        return self._reject_turn(game_id, current_phase,
                                "系统词验证失败，请稍后重试",
                                game.get("history", []))
    
    # 3️⃣ 对比验证
    if self.is_guess_correct(user_guess, target_word):
        # ✅ 用户猜对 - 游戏结束
        now = datetime.now()
        await self.repository.finish_game(game_id, "user_win", now)
        
        # 记录到 history
        final_history = game.get("history", []) + [
            self._make_history_item(
                round_no=len(game.get("history", [])) + 1,
                actor="user",
                turn_type="guess",
                input_text=user_guess,
                answer_text="correct_guess",
                result="user_guessed_correctly",
                source_word_owner="system",
                visible_to_agent=False,
                visible_to_player=True,
            )
        ]
        
        return {
            "game_id": game_id,
            "status": "finished",
            "result": "user_win",
            "message": f"🎉 恭喜你猜对了！目标词是：{target_word}",
            "target_word": target_word,
            "guess": user_guess,
            "history": final_history,
        }
    
    else:
        # ❌ 用户猜错 - 继续游戏
        updated_history = game.get("history", []) + [
            self._make_history_item(
                round_no=len(game.get("history", [])) + 1,
                actor="user",
                turn_type="guess",
                input_text=user_guess,
                answer_text="wrong_guess",
                result="user_guessed_incorrectly",
                source_word_owner="system",
                visible_to_agent=False,
                visible_to_player=True,
            )
        ]
        
        await self.repository.update_game_state(
            game_id,
            history=updated_history,
            summary=self.memory_policy.build_summary(updated_history),
            round_count=len(updated_history),
            phase="user_turn",
            awaiting_judgement=False,
            awaiting_answer=False,
            pending_guess="",
            pending_question="",
            agent_confidence=0.0,
            last_actor="user",
            last_intent="guess",
            game_mode=game.get("metadata", {}).get("game_mode"),
            reasoning_scope=game.get("metadata", {}).get("reasoning_scope"),
            next_actor="user",
            expected_turn_type="guess",
        )
        
        return {
            "game_id": game_id,
            "status": "active",
            "phase": "user_turn",
            "result": "guess_incorrect",
            "message": f"❌ 很遗憾，'{user_guess}' 不是目标词。游戏继续，继续提问或猜测吧！",
            "guess": user_guess,
            "history": updated_history,
        }
```

**关键改动:**
- ✅ 添加猜测词验证（不为空）
- ✅ 添加解密异常处理
- ✅ 完整的 history 记录（猜对和猜错都记录）
- ✅ 游戏结束时返回 `target_word`
- ✅ 猜错时返回新的 `expected_turn_type: "guess"`（提示可继续猜测）

    """处理用户对系统猜测的判定 - 完整版"""
    game_id = game["game_id"]
    pending_guess = game.get("metadata", {}).get("pending_guess", "")

    judgment_text = user_input.strip().lower()

        "是的", "yes", "y", "true", "确实", "同意"

    is_affirmative = (
        any(judgment_text == p for p in affirmative_patterns) or
        any(judgment_text.startswith(p) for p in affirmative_patterns[:6])
    )
