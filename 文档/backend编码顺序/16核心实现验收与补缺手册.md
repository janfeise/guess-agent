```python
class GameService:
	async def submit_turn(self, game_id: str, question: str, mode: str = "agent"):
		game = await self.repository.get_game(game_id)
		if not game:
			raise ValueError(f"Game with id {game_id} not found")

		if game.get("status") == "finished":
			raise ValueError(f"Game with id {game_id} has already finished")

		metadata = game.get("metadata", {})
		phase = metadata.get("phase", "user_turn")
		history = game.get("history", [])

		if phase == "awaiting_judgement":
			return await self._handle_agent_guess_judgement(game, question)

		intent_result = await self.agent.parse_user_intent(question, history)

		if intent_result["intent"] == "invalid":
			return {
				"game_id": game_id,
				"status": game.get("status"),
				"phase": phase,
				"message": "请输入有效问题或猜测",
				"history": history,
			}

		if intent_result["intent"] == "guess":
			return await self._handle_user_guess(game, intent_result)

		return await self._handle_user_question(game, question, history)
```

2. `answer_question()` 负责问答回答
3. `decide_agent_action()` 负责 ask / guess 决策
4. `_parse_json_response()` 负责把模型返回文本转成字典
5. prompt 名称必须和实际文件名一致

#### 建议代码骨架

```python
import json


class GuessAgent:
	def _parse_json_response(self, content: str) -> dict:
		try:
			return json.loads(content)
		except Exception:
			return {
				"intent": "invalid",
				"reason": "模型输出格式不合法",
			}

	async def parse_user_intent(self, user_input: str, history: list[dict]) -> dict:
		messages = [
			{"role": "system", "content": self.prompt_loader.load_prompt("turn_router")},
			{"role": "user", "content": f"请分析用户输入：{user_input}"},
		]
		response = await self.agent_model.ainvoke(messages)
		return self._parse_json_response(response.content)

	async def answer_question(self, question: str, answer: str) -> dict:
		messages = [
			{"role": "system", "content": self.prompt_loader.load_prompt("guess_system")},
			{"role": "system", "content": f"真实答案：{answer}"},
			{"role": "user", "content": question},
		]
		response = await self.agent_model.ainvoke(messages)
		return self._parse_json_response(response.content)

	async def decide_agent_action(self, history: list[dict], summary: str) -> dict:
		messages = [
			{"role": "system", "content": self.prompt_loader.load_prompt("agent_decision")},
			{"role": "user", "content": f"历史对话：{history}\n摘要：{summary}"},
		]
		response = await self.agent_model.ainvoke(messages)
		return self._parse_json_response(response.content)
```

#### 你现在这几个点要特别修正

- `parse_user_intent()` 里的变量名拼错了
- `answer_question()` 不要写成 `self.answer`，要用函数参数 `answer`
- `decide_agent_action()` 的 prompt 名称要和 `prompt/` 目录一致
- `stream_answer()` 目前是纯文本流式，不适合直接当结构化输出用

```python
	self,

	summary: str,
	round_count: int,
```

```text
4. invalid 表示输入无效、闲聊、请求提示、没有游戏意义
5. 只返回 JSON，不要输出解释

输出格式：
{
  "intent": "question | guess | invalid",
  "normalized_text": "",
  "guess_word": "",
  "answer": "yes | no | unknown",
  "confidence": 0.0,
  "reason": ""
}
```

#### 你要确认的点

- 这个 prompt 文件已经存在
- 但 `GuessAgent.parse_user_intent()` 必须真的去加载它
- 不能只写 prompt，不接调用
